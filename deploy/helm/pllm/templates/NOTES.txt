1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "pllm.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "pllm.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "pllm.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "{{ include "pllm.selectorLabels" . }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Admin API is available at:
{{- if .Values.ingress.enabled }}
  Admin endpoints are available on the same ingress at /admin paths
{{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "pllm.fullname" . }} 8081:{{ .Values.service.adminPort }}
  Then visit http://127.0.0.1:8081
{{- end }}

3. Metrics are available at:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "pllm.fullname" . }} 9090:{{ .Values.service.metricsPort }}
  Then visit http://127.0.0.1:9090/metrics

{{- if .Values.dex.enabled }}
4. Dex SSO is available at:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "pllm.fullname" . }}-dex 5556:5556
  Then visit http://127.0.0.1:5556/dex
{{- end }}

5. To get the master key for admin access:
  kubectl get secret --namespace {{ .Release.Namespace }} {{ include "pllm.fullname" . }} -o jsonpath="{.data.master-key}" | base64 --decode

{{- if not .Values.pllm.secrets.openaiApiKey }}
6. Remember to set provider API keys in the secret:
  kubectl patch secret {{ include "pllm.fullname" . }} --namespace {{ .Release.Namespace }} --type='json' -p='[{"op": "replace", "path": "/data/openai-api-key", "value":"<base64-encoded-key>"}]'
{{- end }}
